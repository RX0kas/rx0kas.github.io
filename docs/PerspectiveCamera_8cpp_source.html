25 void PerspectiveCamera::recalculateViewMatrix() {
26 VC_PROFILER_FUNCTION();
27 const float c1 = glm::cos(m_rotation.y); // yaw
28 const float s1 = glm::sin(m_rotation.y);
29 const float c2 = glm::cos(m_rotation.x); // pitch
30 const float s2 = glm::sin(m_rotation.x);
31 const float c3 = glm::cos(m_rotation.z); // roll
32 const float s3 = glm::sin(m_rotation.z);
33
34 const glm::vec3 u{
35 c1 * c3 + s1 * s2 * s3,
36 c2 * s3,
37 c1 * s2 * s3 - c3 * s1
38 };
39
40 const glm::vec3 v{
41 c3 * s1 * s2 - c1 * s3,
42 c2 * c3,
43 c1 * c3 * s2 + s1 * s3
44 };
45
46 const glm::vec3 w{
47 c2 * s1,
48 -s2,
49 c1 * c2
50 };
51
52 m_viewMatrix = glm::mat4{1.0f};
53 m_viewMatrix[0][0] = u.x;
54 m_viewMatrix[1][0] = u.y;
55 m_viewMatrix[2][0] = u.z;
56 m_viewMatrix[0][1] = v.x;
57 m_viewMatrix[1][1] = v.y;
58 m_viewMatrix[2][1] = v.z;
59 m_viewMatrix[0][2] = w.x;
60 m_viewMatrix[1][2] = w.y;
61 m_viewMatrix[2][2] = w.z;
62 m_viewMatrix[3][0] = -glm::dot(u, m_position);
63 m_viewMatrix[3][1] = -glm::dot(v, m_position);
64 m_viewMatrix[3][2] = -glm::dot(w, m_position);
65 }

72 void PerspectiveCamera::recalculateProjectionMatrix() {
73 VC_PROFILER_FUNCTION();
74 const float aspect = Application::instance().window().getAspect();
75 VC_CORE_ASSERT(aspect > std::numeric_limits<float>::epsilon(),"Aspect ratio is zero or invalid");
76
77 const float tanHalfFovy = std::tan(m_fov / 2.f);
78 m_projectionMatrix = glm::mat4{0.0f};
79 m_projectionMatrix[0][0] = 1.f / (aspect * tanHalfFovy);
80 m_projectionMatrix[1][1] = 1.f / (tanHalfFovy);
81 m_projectionMatrix[2][2] = m_camFar / (m_camFar - m_camNear);
82 m_projectionMatrix[2][3] = 1.f;
83 m_projectionMatrix[3][2] = -(m_camFar * m_camNear) / (m_camFar - m_camNear);
84
85 m_projectionMatrix[1][1] *= -1;
86 }
